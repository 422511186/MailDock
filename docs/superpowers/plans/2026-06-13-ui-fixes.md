```typescript
// frontend/src/utils/email.ts
/**
 * 邮箱中间省略：保留开头指定字符 + "..." + "@" 后的域名。
 * 
 * @param email - 邮箱地址
 * @param headLength - 保留开头字符数，默认 8
 * @returns 省略后的邮箱字符串
 * 
 * @example
 * truncateEmail('iog9k1hbmg2q141ftn9zyy9pxn7lzb0p7tb7dakdeagzue8y4@privaterelay.linux.do')
 * // => 'iog9k1hb...privaterelay.linux.do'
 */
export function truncateEmail(email: string, headLength = 8): string {
  // 短邮箱或空值不省略
  if (!email || email.length <= headLength + 15) {
    return email;
  }

  // 找到 @ 符号位置
  const atIndex = email.indexOf('@');
  if (atIndex === -1) {
    // 无 @ 符号，不是邮箱格式，返回原值
    return email;
  }

  // 保留开头 + ... + 域名
  const head = email.slice(0, headLength);
  const domain = email.slice(atIndex + 1);
  return `${head}...${domain}`;
}
```
